# 大程序结构

```c
//add.h
#ifndef ADD_H_
#define ADD_H_

int add(int x,int y);

#endif
```

```c
//add.c
#include "add.h"

int add(int x, int y)
{
    return x + y;
}
```

```c
//main.c
#include <stdio.h>
#include "add.h"

int main(int argc, char const *argv[])
{
    int sum = add(1,2);
    printf("sum == %d\n",sum);
    return 0;
}
```
